151 {
152  QString t_caractere = m_map_IPL[PARAM_CARACTERE_SEPARATEUR].it_mot_courant->to_string();
153  QString t_mot = m_map_IPL[PARAM_MOT_SEPARATEUR].it_mot_courant->to_string();
154  QString t_ligne = m_map_IPL[PARAM_LIGNE_SEPARATEUR].it_mot_courant->to_string();
155 
156  texte t("", t_ligne );
157  QStringList lignes = m_contenu_fichier.split( t_ligne );
158  for ( QStringList::const_iterator it_l = lignes.constBegin(); it_l != lignes.constEnd(); ++it_l )
159  {
160  ligne l("", t_mot );
161  QStringList mots = it_l->split( t_mot );
162  for ( QStringList::const_iterator it_m = mots.constBegin(); it_m != mots.constEnd(); ++it_m )
163  {
164  mot m("", t_caractere );
165  QStringList caracteres = it_m->split( t_caractere );
166  for ( QStringList::const_iterator it_c = caracteres.constBegin(); it_c != caracteres.constEnd(); ++it_c )
167  {
168  if ( it_c->size() != 0 )
169  {
170  base_element e(*it_c);
171  m.push_back(e);
172  }
173  }
174  if ( ! m.empty() )
175  l.ajouter_mot(m);
176  }
177  if ( ! l.empty() )
178  t.ajouter_ligne(l);
179  }
180  textes_out.ajouter_texte(compil.get_configuration(), t);
