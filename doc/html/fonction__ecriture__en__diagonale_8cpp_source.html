158 void fonction_ecriture_en_diagonale::execution_specifique( compilateur &compil, textes & textes_in, textes & textes_out )
159 {
160  QString choix_elements_a_ecrire = m_map_IPL[PARAM_ELEMENTS].it_mot_courant->to_string();
161  QString choix_orientation = m_map_IPL[PARAM_ORIENTATION].it_mot_courant->to_string();
162  int nb_lignes_maximales = m_map_IPL[PARAM_LIGNES].it_caractere_courant->get_entier();
163  bool forcer_rectangularite = m_map_IPL[PARAM_MOTS].it_caractere_courant->get_booleen();
164 
165  if ( m_choix_elements.est_valide( choix_elements_a_ecrire ) &&
166  m_choix_orientation.est_valide( choix_orientation ) )
167  {
168  textes t = textes(textes_in);
169 
170  for ( textes::iterator it_t = t.begin(); it_t != t.end(); ++it_t )
171  {
172  if ( choix_elements_a_ecrire == liste_choix::appliquer_sur_mots )
173  it_t->ecrire_mots_en_diagonale(choix_orientation, nb_lignes_maximales, forcer_rectangularite);
174  else
175  it_t->ecrire_caracteres_en_diagonale(choix_orientation, nb_lignes_maximales, forcer_rectangularite);
176 
177  textes_out.ajouter_texte(compil.get_configuration(), *it_t);
178  }
179  }
