367 bool fenetre_principale::enregistrer_projet(const QString & nom_fichier, projet * p)
368 {
369  QFile file(nom_fichier);
370  if (!file.open(QIODevice::WriteOnly)) {
371  QMessageBox::information(this, tr("Impossible d'ouvrir le fichier."),
372  file.errorString());
373  return false;
374  }
375 
376  p->set_nom_fichier(nom_fichier);
377 
378  QXmlStreamWriter stream(&file);
379  stream.setAutoFormatting(true);
380  stream.writeStartDocument();
381  p->sauvegarder( stream );
382  stream.writeEndDocument();
383 
384  return true;
385 }

672  msgBox.setDefaultButton(QMessageBox::Save);
673  int ret = msgBox.exec();
674 
675  switch (ret) {

677  fermer = enregistrer_projet(p);

680  fermer = true;

683  fermer = false;
