876  msgBox.setStandardButtons(QMessageBox::Save | QMessageBox::Discard | QMessageBox::Cancel);
877  msgBox.setDefaultButton(QMessageBox::Save);
878  int ret = msgBox.exec();
879 
880  switch (ret) {
881  case QMessageBox::Save:
882  fermer = on_enregistrer_projet(p);

895  objet_selectionnable::forcer_deselection();
896
