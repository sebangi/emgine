403 void base_fonction::charger(QXmlStreamReader & xml)
404 {
405  while(xml.readNextStartElement())
406  {
407  if(xml.name() == "nom")
408  {
409  // ignoré
410  QString nom = xml.readElementText();
411  }
412  else if(xml.name() == "niveau_visibilite")
413  {
414  QString niveau_visibilite = xml.readElementText();
415  set_niveau_visibilite( niveau_visibilite.toInt() );
416  }
417  else if(xml.name() == "active")
418  {
419  QString active = xml.readElementText();
420  set_est_active( active.toInt() );
421  }
422  else if(xml.name() == "valeur")
423  {
424  QString valeur = xml.readElementText();
425  if ( get_type() == type_fonction::fonction_source )
426  ((fonction_base_source*)this)->set_string_valeur(valeur);
427  }
428  else if (xml.name() == "objet_selectionnable")
429  objet_selectionnable::charger(xml);
430  else if(xml.name() == "parametres")
431  {
432  charger_parametres(xml);
433  }
434  else
435  {
436  std::cout << "\t\t ignore : " << xml.name().toString().toStdString() << std::endl;
437  xml.skipCurrentElement();
438  }
439  }

496 void base_fonction::algo_IPL_iteration_par_ligne
497 ( type_id_parametre id_param, compilateur &compil, textes & textes_in, textes & textes_out,
498  pf_exec_callback callback )
499 {
500  const textes& t_param = get_textes_parametre(id_param);

502  bool appel = false;
