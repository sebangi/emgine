269 void base_fonction::charger(QXmlStreamReader & xml)
270 {
271  while(xml.readNextStartElement())
272  {
273  if(xml.name() == "nom")
274  {
275  // ignoré
276  QString nom = xml.readElementText();
277  }
278  else if(xml.name() == "niveau_visibilite")
279  {
280  QString niveau_visibilite = xml.readElementText();
281  set_niveau_visibilite( niveau_visibilite.toInt() );
282  }
283  else if(xml.name() == "active")
284  {
285  QString active = xml.readElementText();
286  set_est_active( active.toInt() );
287  }
288  else if(xml.name() == "valeur")
289  {
290  QString valeur = xml.readElementText();
291  if ( get_type() == base_fonction::fonction_source )
292  ((fonction_base_source*)this)->set_string_valeur(valeur);
293  }
294  else if (xml.name() == "objet_selectionnable")
295  objet_selectionnable::charger(xml);
296  else if(xml.name() == "parametres")
297  {
298  charger_parametres(xml);
299  }
300  else
301  {
302  std::cout << "\t\t ignore : " << xml.name().toString().toStdString() << std::endl;
303  xml.skipCurrentElement();
304  }
305  }
