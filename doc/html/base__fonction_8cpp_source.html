391 void base_fonction::charger(QXmlStreamReader & xml)
392 {
393  while(xml.readNextStartElement())
394  {
395  if(xml.name() == "nom")
396  {
397  // ignoré
398  QString nom = xml.readElementText();
399  }
400  else if(xml.name() == "niveau_visibilite")
401  {
402  QString niveau_visibilite = xml.readElementText();
403  set_niveau_visibilite( niveau_visibilite.toInt() );
404  }
405  else if(xml.name() == "active")
406  {
407  QString active = xml.readElementText();
408  set_est_active( active.toInt() );
409  }
410  else if(xml.name() == "valeur")
411  {
412  QString valeur = xml.readElementText();
413  if ( get_type() == type_fonction::fonction_source )
414  ((fonction_base_source*)this)->set_string_valeur(valeur);
415  }
416  else if (xml.name() == "objet_selectionnable")
417  objet_selectionnable::charger(xml);
418  else if(xml.name() == "parametres")
419  {
420  charger_parametres(xml);
421  }
422  else
423  {
424  std::cout << "\t\t ignore : " << xml.name().toString().toStdString() << std::endl;
425  xml.skipCurrentElement();
426  }
427  }
