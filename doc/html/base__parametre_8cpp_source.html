228 bool base_parametre::est_valide(logs_compilation_widget * vue_logs)
229 {
230  bool result = true;
231 
232  type_fonctions actifs;
233  for ( fonctions_iterateur it = m_fonctions.begin(); it != m_fonctions.end(); ++it )
234  if ( (*it)->est_active() )
235  actifs.push_back(*it);
236 
237  if ( actifs.size() == 0 )
238  {
239  vue_logs->ajouter_log
240  ( log_compilation( log_compilation::LOG_ERREUR, this,
241  "Le parametre \"" + m_nom +
242  "\" de la fonction \"" + m_fonction_parent->get_nom() +
243  "\" n'a aucune fonction active") );
244  result = false;
245  }
246  else if ( ! ( actifs.front()->get_type() == type_fonction::fonction_source ) )
247  {
248  vue_logs->ajouter_log
249  ( log_compilation( log_compilation::LOG_ERREUR, this,
250  "Le parametre \"" + m_nom +
251  "\" de la fonction \"" + m_fonction_parent->get_nom() +
252  "\" doit commencer par une fonction source active") );
253  result = false;
254  }
255 
256  for ( fonctions_iterateur it = actifs.begin(); it != actifs.end(); ++it )
257  result = result && (*it)->est_fonction_valide(vue_logs);
258 
259  return result;
260 }

266 void base_parametre::charger(QXmlStreamReader & xml)
267 {
268  while( xml.readNextStartElement() )
269  {
270  if(xml.name() == "nom")
271  {
272  QString nom = xml.readElementText();
273  // on n'en fait rien; c'est seulement pour la lisibilité du fichier
274  }
275  else if(xml.name() == "dans_configuration")
276  {
277  QString dans_configuration = xml.readElementText();
278  if ( dans_configuration.toInt() == 1 )
279  set_mode_configuration_visibilite( CONFIGURATION_VISIBLE );
280  else
281  set_mode_configuration_visibilite( CONFIGURATION_INVISIBLE );
282  }
283  else if (xml.name() == "objet_selectionnable")
284  objet_selectionnable::charger(xml);
285  else if(xml.name() == "fonctions")
286  {
287  charger_fonctions(xml, NULL);
288  }
289  else
290  {
291  std::cout << "\t\t ignore : " << xml.name().toString().toStdString() << std::endl;
292  xml.skipCurrentElement();
293  }
294  }
295 }
