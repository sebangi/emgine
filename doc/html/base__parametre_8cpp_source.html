139 bool base_parametre::est_valide(logs_compilation_widget * vue_logs)
140 {
141  bool result = true;
142 
143  type_fonctions actifs;
144  for ( fonctions_iterateur it = m_fonctions.begin(); it != m_fonctions.end(); ++it )
145  if ( (*it)->est_active() )
146  actifs.push_back(*it);
147 
148  if ( actifs.size() == 0 )
149  {
150  vue_logs->ajouter_log
151  ( log_compilation( log_compilation::LOG_ERREUR, this,
152  "Le parametre \"" + m_nom +
153  "\" de la fonction \"" + m_fonction_parent->get_nom() +
154  "\" n'a aucune fonction active") );
155  result = false;
156  }
157  else if ( ! ( actifs.front()->get_type() == type_fonction::fonction_source ) )
158  {
159  vue_logs->ajouter_log
160  ( log_compilation( log_compilation::LOG_ERREUR, this,
161  "Le parametre \"" + m_nom +
162  "\" de la fonction \"" + m_fonction_parent->get_nom() +
163  "\" doit commencer par une fonction source active") );
164  result = false;
165  }
166 
167  for ( fonctions_iterateur it = actifs.begin(); it != actifs.end(); ++it )
168  result = result && (*it)->est_fonction_valide(vue_logs);
169 
170  return result;
171 }

173 void base_parametre::charger(QXmlStreamReader & xml)
174 {
175  while( xml.readNextStartElement() )
176  {
177  if(xml.name() == "nom")
178  {
179  QString nom = xml.readElementText();
180  // on n'en fait rien; c'est seulement pour la lisibilité du fichier
181  }
182  else if(xml.name() == "dans_configuration")
183  {
184  QString dans_configuration = xml.readElementText();
185  if ( dans_configuration.toInt() == 1 )
186  set_mode_configuration_visibilite( CONFIGURATION_VISIBLE );
187  else
188  set_mode_configuration_visibilite( CONFIGURATION_INVISIBLE );
189  }
190  else if (xml.name() == "objet_selectionnable")
191  objet_selectionnable::charger(xml);
192  else if(xml.name() == "fonctions")
193  {
194  charger_fonctions(xml, NULL);
195  }
196  else
197  {
198  std::cout << "\t\t ignore : " << xml.name().toString().toStdString() << std::endl;
199  xml.skipCurrentElement();
200  }
