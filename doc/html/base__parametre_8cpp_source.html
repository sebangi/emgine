138 bool base_parametre::est_valide(logs_compilation_widget * vue_logs)
139 {
140  bool result = true;
141 
142  type_fonctions actifs;
143  for ( fonctions_iterateur it = m_fonctions.begin(); it != m_fonctions.end(); ++it )
144  if ( (*it)->est_active() )
145  actifs.push_back(*it);
146 
147  if ( actifs.size() == 0 )
148  {
149  vue_logs->ajouter_log
150  ( log_compilation( log_compilation::LOG_ERREUR, this,
151  "Le parametre \"" + m_nom +
152  "\" de la fonction \"" + m_fonction_parent->get_nom() +
153  "\" n'a aucune fonction active") );
154  result = false;
155  }
156  else if ( ! actifs.front()->get_type() == base_fonction::fonction_source )
157  {
158  vue_logs->ajouter_log
159  ( log_compilation( log_compilation::LOG_ERREUR, this,
160  "Le parametre \"" + m_nom +
161  "\" de la fonction \"" + m_fonction_parent->get_nom() +
162  "\" doit commencer par une fonction source active") );
163  result = false;
164  }
165 
166  for ( fonctions_iterateur it = actifs.begin(); it != actifs.end(); ++it )
167  result = result && (*it)->est_fonction_valide(vue_logs);
168 
169  return result;
170 }

172 void base_parametre::charger(QXmlStreamReader & xml)
173 {
174  while( xml.readNextStartElement() )
175  {
176  if(xml.name() == "nom")
177  {
178  QString nom = xml.readElementText();
179  // on n'en fait rien; c'est seulement pour la lisibilité du fichier
180  }
181  else if(xml.name() == "dans_configuration")
182  {
183  QString dans_configuration = xml.readElementText();
184  if ( dans_configuration.toInt() == 1 )
185  set_mode_configuration_visibilite( CONFIGURATION_VISIBLE );
186  else
187  set_mode_configuration_visibilite( CONFIGURATION_INVISIBLE );
188  }
189  else if (xml.name() == "objet_selectionnable")
190  objet_selectionnable::charger(xml);
191  else if(xml.name() == "fonctions")
192  {
193  charger_fonctions(xml, NULL);
194  }
195  else
196  {
197  std::cout << "\t\t ignore : " << xml.name().toString().toStdString() << std::endl;
198  xml.skipCurrentElement();
