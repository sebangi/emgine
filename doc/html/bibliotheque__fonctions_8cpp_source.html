246 bool bibliotheque_fonctions::contient_mot_cle(type_id_fonction id, const QString &cle)
247 {
248  if ( cle.isEmpty() )
249  return true;
250  else
251  {
252  // recherche dans les noms
253  std::map<type_id_fonction, QString>::iterator it_nom = s_fonctions_nom.find(id);
254  if( it_nom != s_fonctions_nom.end() )
255  if ( it_nom->second.contains( cle, Qt::CaseInsensitive ) )
256  return true;
257 
258  // recherche dans l'aide
259  std::map<type_id_fonction, QString>::iterator it_aide = s_fonctions_aide.find(id);
260  if( it_aide != s_fonctions_aide.end() )
261  if ( it_aide->second.contains( cle, Qt::CaseInsensitive ) )
262  return true;
263 
264  // recherche dans les catégories
265  std::map<type_id_fonction, std::set<QString> >::iterator it = s_categories.find(id);
266 
267  if( it != s_categories.end() )
268  for ( std::set<QString>::iterator it_m = it->second.begin(); it_m != it->second.end(); ++it_m )
269  if ( it_m->contains( cle, Qt::CaseInsensitive ) )
270  return true;
271  }
272 
273  return false;
274 }
